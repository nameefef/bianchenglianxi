🐍 Python 基础入门

第1课：Hello World

Python 是一种解释型、面向对象、动态数据类型的高级程序设计语言。

# 第一个 Python 程序
print("Hello, World!")
print("你好，世界！")

第2课：变量和数据类型

Python 中的变量不需要声明类型，直接赋值即可。

# 字符串
name = "小明"

# 整数
age = 18

# 浮点数
height = 1.75

# 布尔值
is_student = True

# 空值
score = None

print(f"姓名：{name}，年龄：{age}岁")

第3课：基本运算

a = 10
b = 3

print(f"加法：{a + b}")      # 13
print(f"减法：{a - b}")      # 7
print(f"乘法：{a * b}")      # 30
print(f"除法：{a / b}")      # 3.333...
print(f"整除：{a // b}")     # 3
print(f"取余：{a % b}")      # 1
print(f"幂运算：{a ** b}")   # 1000

第4课：条件语句

age = 18

if age < 13:
    print("儿童")
elif age < 20:
    print("青少年")  # 输出
else:
    print("成年人")

# 三元表达式
status = "成年" if age >= 18 else "未成年"
print(status)  # 成年

第5课：循环

# for 循环
for i in range(5):
    print(i)  # 0, 1, 2, 3, 4

# while 循环
count = 0
while count < 5:
    print(count)
    count += 1

# 遍历列表
fruits = ["苹果", "香蕉", "橙子"]
for fruit in fruits:
    print(fruit)

⚡ C++ 程序设计

第1课：Hello World

#include <iostream>
using namespace std;

int main() {
    cout << "Hello, World!" << endl;
    cout << "你好，世界！" << endl;
    return 0;
}

第2课：变量和数据类型

#include <iostream>
#include <string>
using namespace std;

int main() {
    // 基本数据类型
    int age = 18;
    double height = 1.75;
    char grade = 'A';
    bool isStudent = true;
    string name = "小明";
    
    cout << "姓名：" << name << endl;
    cout << "年龄：" << age << endl;
    cout << "身高：" << height << "米" << endl;
    
    return 0;
}

第3课：基本运算

#include <iostream>
using namespace std;

int main() {
    int a = 10, b = 3;
    
    cout << "加法：" << a + b << endl;
    cout << "减法：" << a - b << endl;
    cout << "乘法：" << a * b << endl;
    cout << "除法：" << (double)a / b << endl;
    cout << "整除：" << a / b << endl;
    cout << "取余：" << a % b << endl;
    
    return 0;
}

第4课：条件语句

#include <iostream>
using namespace std;

int main() {
    int age = 18;
    
    if (age < 13) {
        cout << "儿童" << endl;
    } else if (age < 20) {
        cout << "青少年" << endl;
    } else {
        cout << "成年人" << endl;
    }
    
    // 三元运算符
    string status = (age >= 18) ? "成年" : "未成年";
    cout << status << endl;
    
    return 0;
}

第5课：循环

#include <iostream>
using namespace std;

int main() {
    // for 循环
    for (int i = 0; i < 5; i++) {
        cout << i << " ";
    }
    cout << endl;
    
    // while 循环
    int count = 0;
    while (count < 5) {
        cout << count << " ";
        count++;
    }
    cout << endl;
    
    // 数组遍历
    int arr[] = {1, 2, 3, 4, 5};
    for (int num : arr) {
        cout << num << " ";
    }
    
    return 0;
}

🗄️ SQL 数据库基础

第1课：SQL 简介

SQL 是结构化查询语言，用于操作数据库。

第2课：查询数据

-- 查询所有数据
SELECT * FROM students;

-- 查询特定列
SELECT name, age FROM students;

-- 条件查询
SELECT * FROM students WHERE age > 18;

第3课：增删改数据

-- 插入数据
INSERT INTO students (name, age) VALUES ('小明', 18);

-- 更新数据
UPDATE students SET age = 19 WHERE name = '小明';

-- 删除数据
DELETE FROM students WHERE name = '小明';
